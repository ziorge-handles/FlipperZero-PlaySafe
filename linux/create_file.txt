REM Linux File Creator
REM Creates a test file in home directory

DELAY 1000
CTRL ALT t
DELAY 1500
STRING echo "BadUSB Test File" > ~/badusb_test.txt
DELAY 500
ENTER
DELAY 500
STRING echo "Created at $(date)" >> ~/badusb_test.txt
DELAY 500
ENTER
DELAY 500
STRING cat ~/badusb_test.txt
DELAY 500
ENTER
DELAY 1000
STRING echo "File created successfully in home directory"
DELAY 500
ENTER
DELAY 1000
STRING exit
DELAY 500
ENTER
