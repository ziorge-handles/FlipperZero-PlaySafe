REM WiFi Credentials Viewer
REM Displays saved WiFi passwords (read-only)
REM Requires admin privileges

DELAY 1000
GUI r
DELAY 500
STRING cmd
DELAY 500
CTRL SHIFT ENTER
DELAY 1000
ALT y
DELAY 500
STRING echo ===== Saved WiFi Networks =====
DELAY 500
ENTER
DELAY 500
STRING netsh wlan show profiles
DELAY 500
ENTER
DELAY 2000
STRING echo.
DELAY 500
ENTER
STRING echo ===== WiFi Passwords =====
DELAY 500
ENTER
DELAY 500
STRING for /f "tokens=2 delims=:" %a in ('netsh wlan show profiles ^| findstr "All User Profile"') do @echo %a && netsh wlan show profile name="%a" key=clear | findstr "Key Content"
DELAY 500
ENTER
DELAY 5000
STRING echo.
DELAY 500
ENTER
STRING echo Display complete. Press any key to close.
DELAY 500
ENTER
STRING pause
DELAY 500
ENTER
DELAY 2000
STRING exit
DELAY 500
ENTER
